### AJAX error handling
```javascript
// MediaAPI.js
const handleFetchErrors = (response) => {
  if (!response.ok) {
    throw Error(response.statusText);
  }
  return response;
}

const getSingleMedia = (id) => {
  return fetch(apiUrl + 'media/' + id)
  .then(handleFetchErrors)
  .then(response => {
    return response.json();
  });
};
```
